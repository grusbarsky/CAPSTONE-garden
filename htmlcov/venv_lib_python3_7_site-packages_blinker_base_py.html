92 def connect(self, receiver, sender=ANY, weak=True): 

93 """Connect *receiver* to signal events sent by *sender*. 

94 

95 :param receiver: A callable. Will be invoked by :meth:`send` with 

96 `sender=` as a single positional argument and any \*\*kwargs that 

97 were provided to a call to :meth:`send`. 

98 

99 :param sender: Any object or :obj:`ANY`, defaults to ``ANY``. 

100 Restricts notifications delivered to *receiver* to only those 

101 :meth:`send` emissions sent by *sender*. If ``ANY``, the receiver 

102 will always be notified. A *receiver* may be connected to 

103 multiple *sender* values on the same Signal through multiple calls 

104 to :meth:`connect`. 

105 

106 :param weak: If true, the Signal will hold a weakref to *receiver* 

107 and automatically disconnect when *receiver* goes out of scope or 

108 is garbage collected. Defaults to True. 

109 

110 """ 

111 receiver_id = hashable_identity(receiver) 

112 if weak: 

113 receiver_ref = reference(receiver, self._cleanup_receiver) 

114 receiver_ref.receiver_id = receiver_id 

115 else: 

116 receiver_ref = receiver 

117 if sender is ANY: 

118 sender_id = ANY_ID 

119 else: 

120 sender_id = hashable_identity(sender) 

121 

122 self.receivers.setdefault(receiver_id, receiver_ref) 

123 self._by_sender[sender_id].add(receiver_id) 

124 self._by_receiver[receiver_id].add(sender_id) 

125 del receiver_ref 

126 

127 if sender is not ANY and sender_id not in self._weak_senders: 

128 # wire together a cleanup for weakref-able senders 

129 try: 

130 sender_ref = reference(sender, self._cleanup_sender) 

131 sender_ref.sender_id = sender_id 

132 except TypeError: 

133 pass 

134 else: 

135 self._weak_senders.setdefault(sender_id, sender_ref) 

136 del sender_ref 

137 

138 # broadcast this connection. if receivers raise, disconnect. 

139 if ('receiver_connected' in self.__dict__ and 

140 self.receiver_connected.receivers): 

141 try: 

142 self.receiver_connected.send(self, 

143 receiver=receiver, 

144 sender=sender, 

145 weak=weak) 

146 except: 

147 self.disconnect(receiver, sender) 

148 raise 

149 if receiver_connected.receivers and self is not receiver_connected: 

150 try: 

151 receiver_connected.send(self, 

152 receiver_arg=receiver, 

153 sender_arg=sender, 

154 weak_arg=weak) 

155 except: 

156 self.disconnect(receiver, sender) 

157 raise 

158 return receiver

241 def send(self, *sender, **kwargs): 

242 """Emit this signal on behalf of *sender*, passing on \*\*kwargs. 

243 

244 Returns a list of 2-tuples, pairing receivers with their return 

245 value. The ordering of receiver notification is undefined. 

246 

247 :param \*sender: Any object or ``None``. If omitted, synonymous 

248 with ``None``. Only accepts one positional argument.

251

254 # used as a keyword argument- i.e. it's an invisible name in the 

255 # function signature. 

256 if len(sender) == 0: 

257 sender = None 

258 elif len(sender) > 1: 

259 raise TypeError('send() accepts only one positional argument, '

261 else: 

262 sender = sender[0] 

263 if not self.receivers: 

264 return []

329 def _disconnect(self, receiver_id, sender_id): 

330 if sender_id == ANY_ID: 

331 if self._by_receiver.pop(receiver_id, False): 

332 for bucket in self._by_sender.values(): 

333 bucket.discard(receiver_id) 

334 self.receivers.pop(receiver_id, None) 

335 else: 

336 self._by_sender[sender_id].discard(receiver_id) 

337 self._by_receiver[receiver_id].discard(sender_id)

344 """Disconnect all receivers from a sender.""" 

345 sender_id = sender_ref.sender_id 

346 assert sender_id != ANY_ID 

347 self._weak_senders.pop(sender_id, None) 

348 for receiver_id in self._by_sender.pop(sender_id, ()):

351 def _cleanup_bookkeeping(self): 

352 """Prune unused sender/receiver bookeeping. Not threadsafe. 

353 

354 Connecting & disconnecting leave behind a small amount of bookeeping 

355 for the receiver and sender values. Typical workloads using Blinker, 

356 for example in most web apps, Flask, CLI scripts, etc., are not 

357 adversely affected by this bookkeeping. 

358 

359 With a long-running Python process performing dynamic signal routing 

360 with high volume- e.g. connecting to function closures, "senders" are 

361 all unique object instances, and doing all of this over and over- you 

362 may see memory usage will grow due to extraneous bookeeping. (An empty 

363 set() for each stale sender/receiver pair.) 

364 

365 This method will prune that bookeeping away, with the caveat that such 

366 pruning is not threadsafe. The risk is that cleanup of a fully 

367 disconnected receiver/sender pair occurs while another thread is 

368 connecting that same pair. If you are in the highly dynamic, unique 

369 receiver/sender situation that has lead you to this method, that 

370 failure mode is perhaps not a big deal for you. 

371 """ 

372 for mapping in (self._by_sender, self._by_receiver): 

373 for _id, bucket in list(mapping.items()): 

374 if not bucket: 

375 mapping.pop(_id, None)
